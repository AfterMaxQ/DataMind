```typescript
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';

export const useThemeStore = defineStore('theme', () => {
  const mode = ref<'dark' | 'light'>('dark');

  const isDark = computed(() => mode.value === 'dark');

  function init() {
    const saved = localStorage.getItem('datamind-theme');
    if (saved === 'dark' || saved === 'light') {
      setTheme(saved);
    } else {
      const prefersLight = window.matchMedia('(prefers-color-scheme: light)').matches;
      setTheme(prefersLight ? 'light' : 'dark');
    }
  }

  function setTheme(newMode: 'dark' | 'light') {
    mode.value = newMode;
    document.documentElement.setAttribute('data-theme', newMode);
    localStorage.setItem('datamind-theme', newMode);
  }

  function toggle() {
    setTheme(mode.value === 'dark' ? 'light' : 'dark');
  }

  return { mode, isDark, init, toggle, setTheme };
});
```

```typescript
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import type { Skill, PhaseInfo, GateStatus } from '@/types';

export const useSessionStore = defineStore('session', () => {
  const sessionId = ref<string | null>(null);
  const currentSkill = ref<Skill | null>(null);
  const currentPhase = ref<PhaseInfo | null>(null);
  const phaseHistory = ref<PhaseInfo[]>([]);
  const gatePending = ref<GateStatus | null>(null);

  const isInGate = computed(() => gatePending.value !== null && !gatePending.value.resolved);
  const currentPhaseName = computed(() => currentPhase.value?.name || '');

  function setSession(id: string) {
    sessionId.value = id;
  }

  function setSkill(skill: Skill | null) {
    currentSkill.value = skill;
  }

  function updatePhase(phase: PhaseInfo) {
    currentPhase.value = phase;
    const existing = phaseHistory.value.findIndex(p => p.name === phase.name);
    if (existing >= 0) {
      phaseHistory.value[existing] = phase;
    } else {
      phaseHistory.value.push(phase);
    }
  }

  function setGatePending(gate: GateStatus | null) {
    gatePending.value = gate;
  }

  async function approveGate(approved: boolean, comment?: string) {
    if (!sessionId.value || !gatePending.value) return;
    try {
      const payload = {
        session_id: sessionId.value,
        gate_name: gatePending.value.gateName,
        decision: { approved, comment }
      };
      const res = await fetch('/skill/gate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
      if (res.ok) {
        if (gatePending.value) {
          gatePending.value.resolved = true;
          gatePending.value.approved = approved;
          gatePending.value.comment = comment;
        }
      }
    } catch (e) {
      console.error('Gate approval failed:', e);
    }
  }

  function reset() {
    sessionId.value = null;
    currentSkill.value = null;
    currentPhase.value = null;
    phaseHistory.value = [];
    gatePending.value = null;
  }

  return {
    sessionId, currentSkill, currentPhase, phaseHistory, gatePending,
    isInGate, currentPhaseName,
    setSession, setSkill, updatePhase, setGatePending, approveGate, reset
  };
});
```

```typescript
  const datasets = ref<Dataset[]>([]);

  const uploadProgress = ref(0);
  const uploadError = ref<string | null>(null);

  const rawDatasets = computed(() => datasets.value.filter(d => d.type === 'raw'));
  const processedDatasets = computed(() => datasets.value.filter(d => d.type === 'processed'));

  async function fetchDatasets() {
    try {
      const res = await fetch('/api/datasets');
      if (res.ok) {
        datasets.value = await res.json();
      }
    } catch (e) {
      console.error('Failed to fetch datasets:', e);
    }
  }

  async function uploadFile(file: File) {
    isUploading.value = true;
    uploadProgress.value = 0;
    uploadError.value = null;
    
    const formData = new FormData();
    formData.append('file', file);

    try {
      // Note: standard fetch doesn't support upload progress. 
      // For a real XHR we'd use XMLHttpRequest, but keeping fetch per spec and simulating progress.
      const interval = setInterval(() => {
        if (uploadProgress.value < 90) uploadProgress.value += 10;
      }, 100);

      const res = await fetch('/upload', {
        method: 'POST',
        body: formData
      });
      
      clearInterval(interval);
      uploadProgress.value = 100;

      if (!res.ok) throw new Error('Upload failed');
      await fetchDatasets();
    } catch (e: unknown) {
      uploadError.value = e instanceof Error ? e.message : 'Unknown error';
    } finally {
      setTimeout(() => {
        isUploading.value = false;
        uploadProgress.value = 0;
      }, 500);
    }
  }

  async function removeDataset(id: string) {
    datasets.value = datasets.value.filter(d => d.id !== id);
  }

  return {
    datasets, rawDatasets, processedDatasets, isUploading, uploadProgress, uploadError,
    fetchDatasets, uploadFile, removeDataset
  };
});
```

```typescript
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import type { ChatMessage, SSEEvent, ToolCall } from '@/types';
import { useSessionStore } from './session';

export const useChatStore = defineStore('chat', () => {
  const messages = ref<ChatMessage[]>([]);
  const isLoading = ref(false);
  const streamingMessageId = ref<string | null>(null);
  const error = ref<string | null>(null);

  const sessionStore = useSessionStore();

  const lastMessage = computed(() => messages.value[messages.value.length - 1] || null);
  const messageCount = computed(() => messages.value.length);

  function generateId() {
    return Math.random().toString(36).substring(2, 9);
  }

  function addSystemMessage(content: string) {
    messages.value.push({ id: generateId(), role: 'system', content, timestamp: Date.now() });
  }

  function clearMessages() {
    messages.value = [];
    error.value = null;
  }

  async function sendMessage(content: string, skill?: string) {
    if (!sessionStore.sessionId) {
      sessionStore.setSession(generateId());
    }

    messages.value.push({
      id: generateId(),
      role: 'user',
      content,
      timestamp: Date.now()
    });

    const assistId = generateId();
    streamingMessageId.value = assistId;
    messages.value.push({
      id: assistId,
      role: 'assistant',
      content: '',
      timestamp: Date.now(),
      isStreaming: true,
      toolCalls: []
    });

    isLoading.value = true;
    error.value = null;

    try {
      const response = await fetch('/chat/stream', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          session_id: sessionStore.sessionId,
          message: content,
          skill
        })
      });

      if (!response.body) throw new Error('No response body');

      const reader = response.body.getReader();
      const decoder = new TextDecoder();
      let buffer = '';

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split('\n');
        buffer = lines.pop() || '';

        for (const line of lines) {
          if (line.startsWith('data: ')) {
            try {
              const event: SSEEvent = JSON.parse(line.slice(6));
              handleSSEEvent(event, assistId);
            } catch (e) {
              console.error('Failed to parse SSE line:', line);
            }
          }
        }
      }
    } catch (e: unknown) {
      error.value = e instanceof Error ? e.message : 'Unknown error';
      addSystemMessage(`Error: ${error.value}`);
    } finally {
      isLoading.value = false;
      streamingMessageId.value = null;
      const msg = messages.value.find(m => m.id === assistId);
      if (msg) msg.isStreaming = false;
    }
  }

  function handleSSEEvent(event: SSEEvent, msgId: string) {
    const msg = messages.value.find(m => m.id === msgId);
    if (!msg) return;

    switch (event.type) {
      case 'token':
        msg.content += event.content;
        break;
      case 'tool_call':
        if (!msg.toolCalls) msg.toolCalls = [];
        msg.toolCalls.push({
          id: generateId(),
          toolName: event.tool,
          args: event.args,
          status: 'running'
        });
        break;
      case 'tool_result':
        if (msg.toolCalls) {
          const tc = msg.toolCalls.find(t => t.toolName === event.tool && t.status === 'running');
          if (tc) {
            tc.status = 'completed';
            tc.result = { success: true, data: event.result };
          }
        }
        break;
      case 'phase':
        sessionStore.updatePhase({ name: event.phase, status: event.status === 'started' ? 'running' : 'completed', startedAt: Date.now() });
        break;
      case 'gate':
        const gate = { gateName: event.gate_name, phase: event.phase, resolved: false };
        sessionStore.setGatePending(gate);
        msg.gateStatus = gate;
        break;
      case 'done':
        msg.isStreaming = false;
        break;
      case 'error':
        addSystemMessage(`System Error: ${event.message}`);
        msg.isStreaming = false;
        break;
    }
  }

  return { messages, isLoading, streamingMessageId, error, lastMessage, messageCount, sendMessage, clearMessages, addSystemMessage };
});
```

### 8. `src/composables/useWebSocket.ts`
```typescript
import { ref, onUnmounted } from 'vue';
import { useSessionStore } from '@/stores/session';
import type { WSEvent, LineageNode, LineageEdge } from '@/types';

// Global state for WS data
export const wsLineageNodes = ref<LineageNode[]>([]);
export const wsLineageEdges = ref<LineageEdge[]>([]);
export const wsDecisions = ref<Array<{phase: string, approved: boolean, comment?: string, timestamp: number}>>([]);

export function useWebSocket() {
  let ws: WebSocket | null = null;
  let reconnectAttempt = 0;
  const maxReconnectDelay = 30000;
  const baseDelay = 1000;
  let reconnectTimeout: ReturnType<typeof setTimeout> | null = null;

  const sessionStore = useSessionStore();

  function connect() {
    const protocol = location.protocol === 'https:' ? 'wss:' : 'ws:';
    const url = `${protocol}//${location.host}/ws`;
    
    ws = new WebSocket(url);

    ws.onopen = () => {
      reconnectAttempt = 0;
    };

    ws.onmessage = (event: MessageEvent) => {
      try {
        const data = JSON.parse(event.data) as WSEvent;
        handleMessage(data);
      } catch (e) {
        console.error('WS parse error:', e);
      }
    };

    ws.onclose = () => {
      scheduleReconnect();
    };

    ws.onerror = () => {
      ws?.close();
    };
  }

  function handleMessage(event: WSEvent) {
    switch (event.type) {
      case 'phase_transition':
        sessionStore.updatePhase({
          name: event.to as string,
          status: 'running',
          startedAt: Date.now()
        });
        break;
      case 'lineage_update':
        if (event.nodes && Array.isArray(event.nodes)) wsLineageNodes.value = event.nodes as LineageNode[];
        if (event.edges && Array.isArray(event.edges)) wsLineageEdges.value = event.edges as LineageEdge[];
        break;
      case 'decision_update':
        wsDecisions.value.unshift({
          phase: event.phase as string,
          approved: event.approved as boolean,
          comment: event.comment as string | undefined,
          timestamp: Date.now()
        });
        if (sessionStore.gatePending && sessionStore.gatePending.phase === event.phase) {
           sessionStore.gatePending.resolved = true;
        }
        break;
      case 'token_stream':
        // Ignored as SSE handles it
        break;
    }
  }

  function scheduleReconnect() {
    if (reconnectTimeout) clearTimeout(reconnectTimeout);
    const delay = Math.min(baseDelay * Math.pow(2, reconnectAttempt), maxReconnectDelay);
    reconnectAttempt++;
    reconnectTimeout = setTimeout(() => {
      connect();
    }, delay);
  }

  function disconnect() {
    if (reconnectTimeout) clearTimeout(reconnectTimeout);
    if (ws) {
      ws.onclose = null;
      ws.close();
    }
  }

  onUnmounted(() => {
    disconnect();
  });

  return { connect, disconnect };
}
```

```vue
        <CodeBlock v-if="part.isCode" :code="part.text" :language="part.lang" />

      </template>

      <!-- Tool Calls -->
      <div v-if="message.toolCalls && message.toolCalls.length > 0" class="tools-container">

      </div>

      <!-- Gate Approval -->
      <GateApproval 
        v-if="message.gateStatus && !message.gateStatus.resolved" 
        :gateStatus="message.gateStatus"
        @approve="(comment) => emit('approve-gate', message.gateStatus!.gateName, true, comment)"
        @reject="(comment) => emit('approve-gate', message.gateStatus!.gateName, false, comment)"
      />

<script setup lang="ts">
import { computed } from 'vue';
import type { ChatMessage } from '@/types';
import CodeBlock from './CodeBlock.vue';
import ToolCallCard from './ToolCallCard.vue';
import GateApproval from './GateApproval.vue';

const props = defineProps<{
  message: ChatMessage;
}>();

const emit = defineEmits<{
  'approve-gate': [gateName: string, approved: boolean, comment?: string];
}>();

// Simple markdown code block parser
const parsedContent = computed(() => {
  const text = props.message.content || '';
  const regex = /```(\w+)?\n([\s\S]*?)```/g;
  const parts = [];
  let lastIndex = 0;
  let match;

  while ((match = regex.exec(text)) !== null) {
    if (match.index > lastIndex) {
      parts.push({ isCode: false, text: text.slice(lastIndex, match.index) });
    }
    parts.push({ isCode: true, lang: match[1] || 'text', text: match[2] });
    lastIndex = regex.lastIndex;
  }
  if (lastIndex < text.length) {
    parts.push({ isCode: false, text: text.slice(lastIndex) });
  }
  
  if (parts.length === 0) return [{ isCode: false, text }];
  return parts;
});

const highlightSkill = (text: string) => {
  return text.replace(/(\/skill\s+[\w-]+)/g, '<span class="skill-cmd">$1</span>');
};
```

```vue
        </svg>
      </button>

          class="autocomplete-item"
          :class="{ selected: index === selectedIndex }"
          @click="selectSkill(skill)"

<script setup lang="ts">
import { ref, computed, nextTick } from 'vue';

const props = defineProps<{
  disabled?: boolean;
  skillNames?: string[];
}>();

const emit = defineEmits<{
  send: [content: string];
}>();

const content = ref('');
const textareaRef = ref<HTMLTextAreaElement | null>(null);
const showAutocomplete = ref(false);
const selectedIndex = ref(0);

const filteredSkills = computed(() => {
  if (!props.skillNames || !content.value.startsWith('/skill ')) return [];
  const search = content.value.slice(7).toLowerCase();
  return props.skillNames.filter(s => s.toLowerCase().includes(search));
});

const handleInput = () => {
  adjustHeight();
  if (content.value.startsWith('/skill ')) {
    showAutocomplete.value = true;
    selectedIndex.value = 0;
  } else {
    showAutocomplete.value = false;
  }
};

const adjustHeight = async () => {
  await nextTick();
  if (textareaRef.value) {
    textareaRef.value.style.height = 'auto';
    const newHeight = Math.min(textareaRef.value.scrollHeight, 120); // max ~6 lines
    textareaRef.value.style.height = `${newHeight}px`;
  }
};

const handleKeydown = (e: KeyboardEvent) => {
  if (showAutocomplete.value && filteredSkills.value.length > 0) {
    if (e.key === 'ArrowDown') {
      e.preventDefault();
      selectedIndex.value = (selectedIndex.value + 1) % filteredSkills.value.length;
      return;
    }
    if (e.key === 'ArrowUp') {
      e.preventDefault();
      selectedIndex.value = (selectedIndex.value - 1 + filteredSkills.value.length) % filteredSkills.value.length;
      return;
    }
    if (e.key === 'Enter') {
      e.preventDefault();
      selectSkill(filteredSkills.value[selectedIndex.value]);
      return;
    }
    if (e.key === 'Escape') {
      showAutocomplete.value = false;
      return;
    }
  }

  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    send();
  }
};

const selectSkill = (skill: string) => {
  content.value = `/skill ${skill} `;
  showAutocomplete.value = false;
  textareaRef.value?.focus();
  adjustHeight();
};

const send = () => {
  if (props.disabled) return;
  const text = content.value.trim();
  if (text) {
    emit('send', text);
    content.value = '';
    showAutocomplete.value = false;
    adjustHeight();
  }
};
```

```vue
  >
    <input 
      type="file" 
      ref="fileInput"

      <svg viewBox="0 0 24 24" width="32" height="32" class="upload-icon">
        <path fill="currentColor" d="M19 13v5c0 1.1-.9 2-2 2H7c-1.1 0-2-.9-2-2v-5H3l8.5-9L20 13h-1zM11 11v6h2v-6h-2z"/>
      </svg>

<script setup lang="ts">
import { ref } from 'vue';

const props = defineProps<{
  isUploading?: boolean;
  uploadProgress?: number;
}>();

const emit = defineEmits<{
  'file-selected': [file: File];
}>();

const isDragOver = ref(false);
const fileInput = ref<HTMLInputElement | null>(null);

const triggerFileSelect = () => {
  if (props.isUploading) return;
  fileInput.value?.click();
};

const handleFileSelect = (e: Event) => {
  const target = e.target as HTMLInputElement;
  if (target.files && target.files.length > 0) {
    emit('file-selected', target.files[0]);
    target.value = ''; // reset
  }
};

const handleDrop = (e: DragEvent) => {
  isDragOver.value = false;
  if (props.isUploading) return;
  if (e.dataTransfer?.files && e.dataTransfer.files.length > 0) {
    emit('file-selected', e.dataTransfer.files[0]);
  }
};
</script>
```

### 19. `src/components/context/LineageGraph.vue`
```vue
<template>
  <div class="graph-container" ref="containerRef">
    <svg ref="svgRef"></svg>
  </div>
</template>

<script setup lang="ts">
import { ref, watch, onMounted, onUnmounted } from 'vue';
import * as d3 from 'd3';
import type { LineageNode, LineageEdge } from '@/types';

const props = defineProps<{
  nodes: LineageNode[];
  edges: LineageEdge[];
}>();

const containerRef = ref<HTMLElement | null>(null);
const svgRef = ref<SVGSVGElement | null>(null);

let simulation: d3.Simulation<d3.SimulationNodeDatum, undefined> | null = null;

// Extend node datum for D3
interface GraphNode extends d3.SimulationNodeDatum, LineageNode {}
interface GraphEdge extends d3.SimulationLinkDatum<GraphNode> {
  source: string | GraphNode;
  target: string | GraphNode;
}

const renderGraph = () => {
  if (!svgRef.value || !containerRef.value) return;

  const width = containerRef.value.clientWidth || 300;
  const height = containerRef.value.clientHeight || 400;

  const svg = d3.select(svgRef.value);
  svg.selectAll('*').remove();

  svg.attr('width', '100%')
     .attr('height', '100%')
     .style('background', 'var(--color-bg-primary)');

  const g = svg.append('g');

  // Zoom behavior
  const zoom = d3.zoom<SVGSVGElement, unknown>()
    .scaleExtent([0.1, 4])
    .on('zoom', (event) => {
      g.attr('transform', event.transform);
    });
  svg.call(zoom);

  // Arrow marker
  svg.append('defs').append('marker')
    .attr('id', 'arrowhead')
    .attr('viewBox', '-0 -5 10 10')
    .attr('refX', 20)
    .attr('refY', 0)
    .attr('orient', 'auto')
    .attr('markerWidth', 6)
    .attr('markerHeight', 6)
    .attr('xoverflow', 'visible')
    .append('svg:path')
    .attr('d', 'M 0,-5 L 10 ,0 L 0,5')
    .attr('fill', 'var(--color-border)')
    .style('stroke', 'none');

  const nodesData: GraphNode[] = props.nodes.map(n => ({ ...n }));
  const edgesData: GraphEdge[] = props.edges.map(e => ({ source: e.source, target: e.target }));

  if (simulation) simulation.stop();

  simulation = d3.forceSimulation<GraphNode>(nodesData)
    .force('link', d3.forceLink<GraphNode, GraphEdge>(edgesData).id(d => d.id).distance(100))
    .force('charge', d3.forceManyBody().strength(-300))
    .force('center', d3.forceCenter(width / 2, height / 2))
    .force('collide', d3.forceCollide().radius(30));

  const link = g.append('g')
    .attr('class', 'links')
    .selectAll('line')
    .data(edgesData)
    .enter().append('line')
    .attr('stroke', 'var(--color-border)')
    .attr('stroke-width', 2)
    .attr('marker-end', 'url(#arrowhead)');

  const nodeGroup = g.append('g')
    .attr('class', 'nodes')
    .selectAll('g')
    .data(nodesData)
    .enter().append('g')
    .call(d3.drag<SVGGElement, GraphNode>()
      .on('start', dragstarted)
      .on('drag', dragged)
      .on('end', dragended) as any); // Cast necessary due to D3 generic typings complexity

  // Node shapes
  nodeGroup.each(function(d) {
    const el = d3.select(this);
    if (d.type === 'dataset') {
      el.append('circle').attr('r', 16).attr('fill', 'var(--color-info)');
    } else if (d.type === 'operation') {
      el.append('rect').attr('width', 32).attr('height', 32).attr('x', -16).attr('y', -16).attr('rx', 4).attr('fill', 'var(--color-warning)');
    } else {
      // result (diamond)
      el.append('polygon').attr('points', '0,-16 16,0 0,16 -16,0').attr('fill', 'var(--color-success)');
    }
  });

  // Labels
  nodeGroup.append('text')
    .text(d => d.label)
    .attr('x', 0)
    .attr('y', 24)
    .attr('text-anchor', 'middle')
    .style('font-size', '12px')
    .style('fill', 'var(--color-text-primary)');

  simulation.on('tick', () => {
    link
      .attr('x1', d => (d.source as GraphNode).x!)
      .attr('y1', d => (d.source as GraphNode).y!)
      .attr('x2', d => (d.target as GraphNode).x!)
      .attr('y2', d => (d.target as GraphNode).y!);

    nodeGroup.attr('transform', d => `translate(${d.x},${d.y})`);
  });

  function dragstarted(event: any, d: GraphNode) {
    if (!event.active && simulation) simulation.alphaTarget(0.3).restart();
    d.fx = d.x;
    d.fy = d.y;
  }
  function dragged(event: any, d: GraphNode) {
    d.fx = event.x;
    d.fy = event.y;
  }
  function dragended(event: any, d: GraphNode) {
    if (!event.active && simulation) simulation.alphaTarget(0);
    d.fx = null;
    d.fy = null;
  }
};

watch(() => props.nodes, renderGraph, { deep: true });
watch(() => props.edges, renderGraph, { deep: true });

onMounted(() => {
  renderGraph();
  window.addEventListener('resize', renderGraph);
});

onUnmounted(() => {
  if (simulation) simulation.stop();
  window.removeEventListener('resize', renderGraph);
});
</script>

  width: 100%;
  min-height: 300px;
  flex: 1;
  overflow: hidden;
}

:deep(.nodes g) {
  cursor: grab;
}

:deep(.nodes g:active) {
  cursor: grabbing;
}
</style>
```
