```typescript
export type MessageRole = 'user' | 'assistant' | 'system';

export interface ToolResult {
  success: boolean;
  data?: unknown;
  error?: string;
}

// Struct representing an individual tool execution stage

  id: string;
  toolName: string;
  args: Record<string, unknown>;
  result?: ToolResult;
  status: ToolCallStatus;
}

// Manual gate approval properties
export interface GateStatus {
  gateName: string;
  phase: string;
  resolved: boolean;
  approved?: boolean;
  comment?: string;
}

// Chat message format stored in Chat state
export interface ChatMessage {
  id: string;
  role: MessageRole;
  content: string;
  timestamp: number;
  toolCalls?: ToolCall[];
  gateStatus?: GateStatus;
  isStreaming?: boolean;
}

export type SSEEvent =
  | { type: 'token'; content: string }
  | { type: 'tool_call'; tool: string; args: Record<string, unknown> }
  | { type: 'tool_result'; tool: string; result: Record<string, unknown> }
  | { type: 'phase'; phase: string; status: 'started' | 'completed' }
```

```typescript
import { defineStore } from 'pinia';

export const useThemeStore = defineStore('theme', {
  state: () => ({
    mode: 'dark' as 'dark' | 'light',
  }),
  getters: {
    isDark(): boolean {
      return this.mode === 'dark';
    },
  },
  actions: {
    init(): void {
      const savedTheme = localStorage.getItem('datamind-theme') as 'dark' | 'light' | null;
      if (savedTheme === 'dark' || savedTheme === 'light') {
        this.setTheme(savedTheme);
      } else {
        const prefersLight = window.matchMedia('(prefers-color-scheme: light)').matches;
        this.setTheme(prefersLight ? 'light' : 'dark');
      }
    },
    toggle(): void {
      this.setTheme(this.mode === 'dark' ? 'light' : 'dark');
    },
    setTheme(mode: 'dark' | 'light'): void {
      this.mode = mode;
      localStorage.setItem('datamind-theme', mode);
      const root = document.documentElement;
      if (mode === 'light') {
        root.setAttribute('data-theme', 'light');
      } else {
        root.removeAttribute('data-theme');
      }
    },
  },
});
```

### 5. `src/stores/session.ts` — Session & Workflow Execution Store

```typescript
import { defineStore } from 'pinia';
import type { Skill, PhaseInfo, GateStatus } from '../types';

export const useSessionStore = defineStore('session', {
  state: () => ({
    sessionId: null as string | null,
    currentSkill: null as Skill | null,
    currentPhase: null as PhaseInfo | null,
    phaseHistory: [] as PhaseInfo[],
    gatePending: null as GateStatus | null,
  }),
  getters: {
    isInGate(): boolean {
      return this.gatePending !== null && !this.gatePending.resolved;
    },
    currentPhaseName(): string {
      return this.currentPhase ? this.currentPhase.name : 'idle';
    },
  },
  actions: {
    setSession(id: string): void {
      this.sessionId = id;
    },
    setSkill(skill: Skill | null): void {
      this.currentSkill = skill;
    },
    updatePhase(phase: PhaseInfo): void {
      this.currentPhase = phase;
      const index = this.phaseHistory.findIndex(p => p.name === phase.name);
      if (index !== -1) {
        this.phaseHistory[index] = phase;
      } else {
        this.phaseHistory.push(phase);
      }
    },
    setGatePending(gate: GateStatus | null): void {
      this.gatePending = gate;
    },
    async approveGate(approved: boolean, comment?: string): Promise<void> {
      if (!this.sessionId || !this.gatePending) return;
      
      const payload = {
        session_id: this.sessionId,
        gate_name: this.gatePending.gateName,
        decision: { approved, comment },
      };

      try {
        const response = await fetch('/skill/gate', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload),
        });
        if (!response.ok) {
          throw new Error('Gate response was not stored successfully');
        }
        
        if (this.gatePending) {
          this.gatePending.resolved = true;
          this.gatePending.approved = approved;
          this.gatePending.comment = comment;
        }
        this.gatePending = null;
      } catch (err) {
        console.error('Error recording manual gate status decision:', err);
        throw err;
      }
    },
    reset(): void {
      this.sessionId = null;
      this.currentSkill = null;
      this.currentPhase = null;
      this.phaseHistory = [];
      this.gatePending = null;
    },
  },
});
```

### 7. `src/stores/chat.ts` — Live Chat Session & SSE Controller Store

```typescript
import { defineStore } from 'pinia';
import type { ChatMessage, ToolCall, SSEEvent } from '../types';
import { useSessionStore } from './session';

export const useChatStore = defineStore('chat', {
  state: () => ({
    messages: [] as ChatMessage[],
    isLoading: false,
    streamingMessageId: null as string | null,
    error: null as string | null,
  }),
  getters: {
    lastMessage(): ChatMessage | null {
      return this.messages.length > 0 ? this.messages[this.messages.length - 1] : null;
    },
    messageCount(): number {
      return this.messages.length;
    },
  },
  actions: {
    clearMessages(): void {
      this.messages = [];
    },
    addSystemMessage(content: string): void {
      this.messages.push({
        id: `sys-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
        role: 'system',
        content,
        timestamp: Date.now(),
      });
    },
    async sendMessage(content: string, skill?: string): Promise<void> {
      this.isLoading = true;
      this.error = null;

      const userMsgId = `user-${Date.now()}`;
      this.messages.push({
        id: userMsgId,
        role: 'user',
        content,
        timestamp: Date.now(),
      });

      const assistantMsgId = `assistant-${Date.now()}`;
      this.streamingMessageId = assistantMsgId;
      
      this.messages.push({
        id: assistantMsgId,
        role: 'assistant',
        content: '',
        timestamp: Date.now(),
        toolCalls: [],
        isStreaming: true,
      });

      const sessionStore = useSessionStore();
      const sId = sessionStore.sessionId || `session-${Date.now()}`;
      sessionStore.setSession(sId);

      try {
        const response = await fetch('/chat/stream', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            session_id: sId,
            message: content,
            skill: skill || undefined,
          }),
        });

        if (!response.ok) {
          throw new Error(`Streaming failed: HTTP Status ${response.status}`);
        }

        const reader = response.body!.getReader();
        const decoder = new TextDecoder();
        let buffer = '';

        while (true) {
          const { done, value } = await reader.read();
          if (done) break;
          buffer += decoder.decode(value, { stream: true });
          const lines = buffer.split('\n');
          buffer = lines.pop() || '';

          for (const line of lines) {
            const cleanLine = line.trim();
            if (!cleanLine) continue;
            
            if (cleanLine.startsWith('data: ')) {
              const rawData = cleanLine.slice(6).trim();
              if (!rawData) continue;
              
              try {
                const event: SSEEvent = JSON.parse(rawData);
                this.handleIncomingSSEEvent(event);
              } catch (e) {
                console.warn('Skipping unparsable SSE frame:', cleanLine, e);
              }
            }
          }
        }
      } catch (err: unknown) {
        const message = err instanceof Error ? err.message : 'An error occurred';
        this.error = message;
        this.addSystemMessage(`Connection Failure: ${message}`);
        
        // Remove or clean up the streaming state if a failure occurs
        const activeMsg = this.messages.find(m => m.id === assistantMsgId);
        if (activeMsg) {
          activeMsg.isStreaming = false;
        }
      } finally {
        this.isLoading = false;
        this.streamingMessageId = null;
        const finalMsg = this.messages.find(m => m.id === assistantMsgId);
        if (finalMsg) {
          finalMsg.isStreaming = false;
        }
      }
    },

    handleIncomingSSEEvent(event: SSEEvent): void {
      if (!this.streamingMessageId) return;
      const currentMsg = this.messages.find(m => m.id === this.streamingMessageId);
      if (!currentMsg) return;

      const sessionStore = useSessionStore();

      switch (event.type) {
        case 'token':
          currentMsg.content += event.content;
          break;

        case 'tool_call':
          if (!currentMsg.toolCalls) {
            currentMsg.toolCalls = [];
          }
          currentMsg.toolCalls.push({
            id: `tool-${Date.now()}-${Math.random().toString(36).substr(2, 5)}`,
            toolName: event.tool,
            args: event.args,
            status: 'running',
          });
          break;

        case 'tool_result':
          if (currentMsg.toolCalls) {
            const activeTool = currentMsg.toolCalls.find(
              t => t.toolName === event.tool && (t.status === 'running' || t.status === 'pending')
            );
            if (activeTool) {
              activeTool.status = 'completed';
              activeTool.result = {
                success: true,
                data: event.result,
              };
            }
          }
          break;

        case 'phase':
          sessionStore.updatePhase({
            name: event.phase,
            status: event.status === 'started' ? 'running' : 'completed',
            startedAt: event.status === 'started' ? Date.now() : undefined,
          });
          break;

        case 'gate':
          const gateStatusStruct = {
            gateName: event.gate_name,
            phase: event.phase,
            resolved: false,
          };
          currentMsg.gateStatus = gateStatusStruct;
          sessionStore.setGatePending(gateStatusStruct);
          break;

        case 'done':
          currentMsg.isStreaming = false;
          break;

        case 'error':
          this.addSystemMessage(`Error Event: ${event.message}`);
          break;
      }
    },
  },
});
```

### 8. `src/composables/useWebSocket.ts` — Realtime WebSockets Orchestrator

```typescript
import { ref, onUnmounted } from 'vue';
import { useSessionStore } from '../stores/session';
import type { WSEvent, LineageNode, LineageEdge } from '../types';

export function useWebSocket(
  onLineageUpdate?: (nodes: LineageNode[], edges: LineageEdge[]) => void,
  onDecisionUpdate?: (decision: { phase: string; approved: boolean; comment?: string; timestamp: number }) => void
) {
  let ws: WebSocket | null = null;
  let reconnectAttempt = 0;
  const baseDelay = 1000;
  const maxReconnectDelay = 30000;
  const isConnected = ref(false);
  let timerId: number | null = null;

  const connect = (): void => {
    const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
    const host = window.location.host;
    const wsUrl = `${protocol}//${host}/ws`;

    ws = new WebSocket(wsUrl);

    ws.onopen = () => {
      isConnected.value = true;
      reconnectAttempt = 0;
      if (timerId) {
        clearTimeout(timerId);
        timerId = null;
      }
    };

    ws.onmessage = (event: MessageEvent) => {
      try {
        const payload: WSEvent = JSON.parse(event.data);
        handleMessage(payload);
      } catch (err) {
        console.warn('Ignoring invalid WebSocket payload format:', err);
      }
    };

    ws.onclose = () => {
      isConnected.value = false;
      scheduleReconnect();
    };

    ws.onerror = (e) => {
      console.error('WebSocket connection encountered errors:', e);
      ws?.close();
    };
  };

  const disconnect = (): void => {
    if (timerId) {
      clearTimeout(timerId);
      timerId = null;
    }
    if (ws) {
      ws.onclose = null; // Prevent triggers in callback loop
      ws.close();
      ws = null;
    }
    isConnected.value = false;
  };

  const handleMessage = (data: WSEvent): void => {
    const sessionStore = useSessionStore();

    switch (data.type) {
      case 'phase_transition':
        sessionStore.updatePhase({
          name: data.to as string,
          status: 'running',
          startedAt: Date.now(),
        });
        break;

      case 'lineage_update':
        if (onLineageUpdate) {
          const nodes = data.nodes as LineageNode[];
          const edges = data.edges as LineageEdge[];
          onLineageUpdate(nodes, edges);
        }
        break;

      case 'decision_update':
        if (onDecisionUpdate) {
          onDecisionUpdate({
            phase: data.phase as string,
            approved: data.approved as boolean,
            comment: data.comment as string | undefined,
            timestamp: Date.now(),
          });
        }
        break;
        
      case 'token_stream':
        // Safe to ignore as SSE processes continuous text generation.
        break;
    }
  };

  const scheduleReconnect = (): void => {
    if (timerId !== null) return;

    const delay = Math.min(baseDelay * Math.pow(2, reconnectAttempt), maxReconnectDelay);
    reconnectAttempt++;

    timerId = window.setTimeout(() => {
      timerId = null;
      connect();
    }, delay);
  };

  onUnmounted(() => {
    disconnect();
  });

  return {
    isConnected,
    connect,
    disconnect,
  };
}
```

```vue
<script setup lang="ts">
import { computed } from 'vue';
import type { ChatMessage } from '../../types';

import GateApproval from './GateApproval.vue';

}>();

const emit = defineEmits<{

}>();

const isUser = computed(() => props.message.role === 'user');
const isSystem = computed(() => props.message.role === 'system');
const isAssistant = computed(() => props.message.role === 'assistant');

const containerClass = computed(() => {
  return {
    'user-bubble': isUser.value,
    'assistant-bubble': isAssistant.value,
    'system-bubble': isSystem.value,
  };
});

interface ContentPart {
  type: 'text' | 'code';
  content: string;
  language?: string;
}

// Split streaming text structures into blocks of Markdown blocks and standard outputs
const messageParts = computed<ContentPart[]>(() => {
  const parts: ContentPart[] = [];
  const content = props.message.content;
  const codeRegex = /```(\w*)\n([\s\S]*?)(?:```|$)/g;
  
  let lastIndex = 0;
  let match;

  while ((match = codeRegex.exec(content)) !== null) {
    if (match.index > lastIndex) {
      parts.push({
        type: 'text',
        content: content.substring(lastIndex, match.index),
      });
    }
    parts.push({
      type: 'code',
      content: match[2],
      language: match[1] || 'python',
    });
    lastIndex = codeRegex.lastIndex;
  }

  if (lastIndex < content.length) {
    parts.push({
      type: 'text',
      content: content.substring(lastIndex),
    });
  }

  return parts;
});

const formattedText = (text: string): string => {
  // Highlights instances of skills keywords like "/skill data-cleaning"
  return text.replace(
    /(\/skill\s+[\w-]+)/g,
    '<span class="skill-cmd">$1</span>'
  );
};

const handleApproval = (approved: boolean, comment?: string) => {
  if (props.message.gateStatus) {
    emit('approve-gate', props.message.gateStatus.gateName, approved, comment);
  }
};
```

```vue
<script setup lang="ts">
import { ref, watch, nextTick, computed } from 'vue';

const props = withDefaults(
  defineProps<{
    disabled?: boolean;
    skillNames?: string[];
  }>(),
  {
    disabled: false,
    skillNames: () => [],
  }
);

  send: [content: string];
}>();

const inputVal = ref('');
const textareaRef = ref<HTMLTextAreaElement | null>(null);
const autocompleteIndex = ref(0);
const showAutocomplete = ref(false);

const currentWordContext = computed(() => {
  const words = inputVal.value.split(/\s+/);
  return words[words.length - 1];
});

const isTriggeringSkill = computed(() => {
  return inputVal.value.trim().startsWith('/skill');
});

const filteredSkills = computed(() => {
  if (!isTriggeringSkill.value) return [];
  const keyword = currentWordContext.value.toLowerCase().replace(/^\/skill/, '');
  return props.skillNames.filter(s => s.toLowerCase().includes(keyword));
});

const hasMatches = computed(() => {
  return filteredSkills.value.length > 0;
});

watch(inputVal, () => {
  adjustHeight();
  
  if (isTriggeringSkill.value && hasMatches.value) {
    showAutocomplete.value = true;
  } else {
    showAutocomplete.value = false;
  }
});

const adjustHeight = () => {
  nextTick(() => {
    if (textareaRef.value) {
      textareaRef.value.style.height = 'auto';
      const scrollHeight = textareaRef.value.scrollHeight;
      textareaRef.value.style.height = `${Math.min(Math.max(scrollHeight, 40), 150)}px`;
    }
  });
};

const handleKeyDown = (e: KeyboardEvent) => {
  if (showAutocomplete.value) {
    if (e.key === 'ArrowDown') {
      e.preventDefault();
      autocompleteIndex.value = (autocompleteIndex.value + 1) % filteredSkills.value.length;
    } else if (e.key === 'ArrowUp') {
      e.preventDefault();
      autocompleteIndex.value = (autocompleteIndex.value - 1 + filteredSkills.value.length) % filteredSkills.value.length;
    } else if (e.key === 'Enter') {
      e.preventDefault();
      applyAutocomplete(filteredSkills.value[autocompleteIndex.value]);
    } else if (e.key === 'Escape') {
      showAutocomplete.value = false;
    }
    return;
  }

  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    handleSubmit();
  }
};

const applyAutocomplete = (skill: string) => {
  inputVal.value = `/skill ${skill} `;
  showAutocomplete.value = false;
  autocompleteIndex.value = 0;
  nextTick(() => {
    textareaRef.value?.focus();
  });
};

const handleSubmit = () => {
  if (props.disabled || !inputVal.value.trim()) return;
  emit('send', inputVal.value.trim());
  inputVal.value = '';
  nextTick(() => {
    if (textareaRef.value) {
      textareaRef.value.style.height = '40px';
    }
  });
};
</script>
```

```vue
  const gContainer = svg.append('g');

  // Inject drag and zoom controls mapping onto the viewport container
  const zoomBehavior = d3.zoom<SVGSVGElement, unknown>()
    .scaleExtent([0.2, 3])
    .on('zoom', (event) => {
      gContainer.attr('transform', event.transform);
    });

  svg.call(zoomBehavior);

  // Define Arrowhead definitions to avoid overlapping connectors
  svg.append('defs').append('marker')
    .attr('id', 'arrow')
    .attr('viewBox', '0 -5 10 10')
    .attr('refX', 22)
    .attr('refY', 0)
    .attr('markerWidth', 6)
    .attr('markerHeight', 6)
    .attr('orient', 'auto')
    .append('path')
    .attr('fill', 'var(--color-text-tertiary)')
    .attr('d', 'M0,-5L10,0L0,5');

  const nodesData: D3LineageNode[] = props.nodes.map(n => ({ ...n }));
  const edgesData: D3LineageEdge[] = props.edges.map(e => ({ ...e }));

  simulation = d3.forceSimulation<D3LineageNode, D3LineageEdge>(nodesData)
    .force('link', d3.forceLink<D3LineageNode, D3LineageEdge>(edgesData).id(d => d.id).distance(90))
    .force('charge', d3.forceManyBody().strength(-200))
    .force('center', d3.forceCenter(width / 2, height / 2))
    .force('collision', d3.forceCollide().radius(25));

  // Render edge elements
  const link = gContainer.append('g')
    .selectAll('line')
    .data(edgesData)
    .enter()
    .append('line')
    .attr('stroke', 'var(--color-border)')
    .attr('stroke-width', 1.5)
    .attr('marker-end', 'url(#arrow)');

  // Render node element nodes
  const node = gContainer.append('g')
    .selectAll('.node-group')
    .data(nodesData)
    .enter()
    .append('g')
    .attr('class', 'node-group')
    .call(d3.drag<SVGGElement, D3LineageNode>()
      .on('start', dragStarted)
      .on('drag', dragged)
      .on('end', dragEnded)
    );

  // Inject node visual shape bases depending on structural dataset type
  node.each(function (d) {
    const group = d3.select(this);
    if (d.type === 'dataset') {
      group.append('circle')
        .attr('r', 16)
        .attr('fill', 'var(--color-accent)')
        .attr('stroke', 'var(--color-bg-secondary)')
        .attr('stroke-width', 2);
    } else if (d.type === 'operation') {
      group.append('rect')
        .attr('width', 30)
        .attr('height', 16)
        .attr('x', -15)
        .attr('y', -8)
        .attr('rx', 4)
        .attr('fill', 'var(--color-warning)')
        .attr('stroke', 'var(--color-bg-secondary)')
        .attr('stroke-width', 2);
    } else {
      group.append('polygon')
        .attr('points', '0,-16 16,0 0,16 -16,0')
        .attr('fill', 'var(--color-success)')
        .attr('stroke', 'var(--color-bg-secondary)')
        .attr('stroke-width', 2);
    }
  });

  // Inject nodes description labels
  node.append('text')
    .attr('text-anchor', 'middle')
    .attr('y', 28)
    .attr('font-size', '10px')
    .attr('font-family', 'var(--font-sans)')
    .attr('fill', 'var(--color-text-secondary)')
    .text(d => d.label);

  simulation.on('tick', () => {
    link
      .attr('x1', d => (d.source as D3LineageNode).x ?? 0)
      .attr('y1', d => (d.source as D3LineageNode).y ?? 0)
      .attr('x2', d => (d.target as D3LineageNode).x ?? 0)
      .attr('y2', d => (d.target as D3LineageNode).y ?? 0);

    node.attr('transform', d => `translate(${d.x ?? 0}, ${d.y ?? 0})`);
  });

  function dragStarted(event: d3.D3DragEvent<SVGGElement, D3LineageNode, unknown>, d: D3LineageNode) {
    if (!event.active) simulation?.alphaTarget(0.3).restart();
    d.fx = d.x;
    d.fy = d.y;
  }

  function dragged(event: d3.D3DragEvent<SVGGElement, D3LineageNode, unknown>, d: D3LineageNode) {
    d.fx = event.x;
    d.fy = event.y;
  }

  function dragEnded(event: d3.D3DragEvent<SVGGElement, D3LineageNode, unknown>, d: D3LineageNode) {
    if (!event.active) simulation?.alphaTarget(0);
    d.fx = null;
    d.fy = null;
  }
};

watch(
  () => [props.nodes, props.edges],
  () => {
    renderGraph();
  },
  { deep: true }
);

onMounted(() => {
  renderGraph();
  window.addEventListener('resize', renderGraph);
});

onUnmounted(() => {
  if (simulation) {
    simulation.stop();
  }
  window.removeEventListener('resize', renderGraph);
});
</script>

<template>

    <svg ref="svgContainerRef" class="lineage-svg"></svg>
  </div>
</template>
```

### 24. `src/App.vue` — Main Application Layout

```vue
<script setup lang="ts">
import { ref, onMounted } from 'vue';
import { useWebSocket } from './composables/useWebSocket';
import AppHeader from './components/layout/AppHeader.vue';
import DataSidebar from './components/data/DataSidebar.vue';
import ChatPanel from './components/chat/ChatPanel.vue';
import ContextPanel from './components/context/ContextPanel.vue';
import type { LineageNode, LineageEdge, DecisionRecord } from './types';

const lineageNodes = ref<LineageNode[]>([]);
const lineageEdges = ref<LineageEdge[]>([]);
const decisionsLog = ref<DecisionRecord[]>([]);

// Event hooks triggered on live updates sent over WebSockets
const handleLineageUpdate = (nodes: LineageNode[], edges: LineageEdge[]) => {
  lineageNodes.value = nodes;
  lineageEdges.value = edges;
};

const handleDecisionUpdate = (decision: DecisionRecord) => {
  decisionsLog.value.unshift(decision);
};

const { connect, disconnect } = useWebSocket(
  handleLineageUpdate,
  handleDecisionUpdate
);

onMounted(() => {
  connect();
  
  // Set default lineage diagram when loading application if API hasn't delivered yet
  if (lineageNodes.value.length === 0) {
    lineageNodes.value = [
      { id: '1', label: 'customer_orders.csv', type: 'dataset' },
      { id: '2', label: 'filtering outlier rows', type: 'operation' },
      { id: '3', label: 'cleansed_table.parquet', type: 'result' }
    ];
    lineageEdges.value = [
      { source: '1', target: '2' },
      { source: '2', target: '3' }
    ];
  }
});
</script>

<template>
  <div class="app-shell-grid">
    <AppHeader />
    <main class="viewport-contents">
      <DataSidebar />
      <div class="center-content-wrapper">
        <ChatPanel />
      </div>
      <ContextPanel 
        :nodes="lineageNodes" 
        :edges="lineageEdges" 
        :decisions="decisionsLog"
      />
    </main>
  </div>
</template>
```
